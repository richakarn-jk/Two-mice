// ฟังก์ชันสำหรับเข้าสู่ระบบ (อัปเดตใหม่ ทะลวงฐานข้อมูลว่าง) async function handleLogin() { // ดึงค่าจากช่องกรอก ID และ Password (แก้ ID ของ input ให้ตรงกับ HTML ของคุณนะครับ) const loginId = document.getElementById('loginIdInput').value.trim(); const loginPass = document.getElementById('loginPassInput').value.trim(); if (!loginId || !loginPass) { alert("กรุณากรอกรหัสพนักงานและรหัสผ่านให้ครบถ้วนครับ"); return; } // ========================================== // กุญแจฉุกเฉิน: Super Admin (เข้าได้แน่นอน 100%) // ========================================== if (loginId === '00' && loginPass === '111') { // บันทึกสถานะว่าเข้าสู่ระบบแล้วลงในเครื่อง const adminData = { id: '00', name: 'Super Admin', role: 'admin', permissions: ['all'] // ให้สิทธิ์ทุกอย่าง }; localStorage.setItem('twomice_current_user', JSON.stringify(adminData)); alert("ยินดีต้อนรับ Super Admin!"); // สั่งซ่อนหน้า Login และแสดงหน้าจอหลัก (แก้ ID ให้ตรงกับ HTML ของคุณ) document.getElementById('loginScreen').style.display = 'none'; document.getElementById('mainPosScreen').style.display = 'block'; // หรือ id ของหน้าจอหลักคุณ // โหลดข้อมูลต่างๆ เพื่อเตรียมใช้งาน // loadInitialData(); return; } // ========================================== // สำหรับพนักงานคนอื่นๆ (ต้องไปเช็คใน Firebase) // ========================================== try { if (typeof db === 'undefined') { alert("ยังไม่ได้เชื่อมต่อฐานข้อมูล ไม่สามารถตรวจสอบรหัสพนักงานอื่นได้ (แต่ใช้ 00/111 เข้าได้ครับ)"); return; } // ค้นหาพนักงานในคอลเลกชัน 'employees' const snapshot = await db.collection('employees').where('id', '==', loginId).get(); if (snapshot.empty) { alert("ไม่พบรหัสพนักงานนี้ในระบบครับ"); return; } const employeeData = snapshot.docs[0].data(); // ตรวจสอบรหัสผ่าน if (employeeData.password !== loginPass) { alert("รหัสผ่านไม่ถูกต้องครับ"); return; } // ถ้ารหัสถูก ให้บันทึกข้อมูลและเข้าสู่ระบบ localStorage.setItem('twomice_current_user', JSON.stringify(employeeData)); alert(`ยินดีต้อนรับ ${employeeData.name}`); // สั่งซ่อนหน้า Login และแสดงหน้าจอหลัก document.getElementById('loginScreen').style.display = 'none'; document.getElementById('mainPosScreen').style.display = 'block'; } catch (error) { console.error("Login Error: ", error); alert("เกิดข้อผิดพลาดในการตรวจสอบข้อมูล กรุณาลองใหม่อีกครั้ง"); } }
